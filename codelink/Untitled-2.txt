#include <stdio.h>

int main() {
    int rows, cols;

    printf("Enter number of rows: ");
    scanf("%d", &rows);

    printf("Enter number of columns: ");
    scanf("%d", &cols);

    int arr[rows][cols];

    // Input elements
    printf("Enter the elements of the matrix:\n");
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            printf("Element [%d][%d]: ", i, j);
            scanf("%d", &arr[i][j]);
        }
    }
    int sum = 0;

    // Calculate row sums
    for (int i = 0; i < rows; i++) {
        

        for (int j = 0; j < cols; j++) {
            sum += arr[i][j];
        }

     
    }

    printf("sum = %d", sum);

    return 0;
}
