#include <stdio.h>

int main(){
    
    int size;
    
    printf("Enter size: ");
    scanf("%d", &size);
    
    int arr[size];
    
    for(int i = 0; i < size ; i++){
        
        scanf("%d", &arr[i]);
        
    }
    
    int temp[size];
    int index = 0;
    
    for(int i = 0 ; i < size ; i++){
        if(arr[i] < 0){
            temp[index++] = arr[i];
        }
    }
    
    for(int i = 0; i < size ; i++){
        if(arr[i] >= 0){
            temp[index++] = arr[i];
        }
    }
    
    for(int i = 0; i < size ; i++){
        
        arr[i] = temp[i];
    }
    
    for(int i = 0; i < size; i++){
        printf("%d ", arr[i]);
    }
    
}
